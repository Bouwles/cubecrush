document.addEventListener('DOMContentLoaded', () => { const boardSize = 8; let board = []; let score = 0; let highScore = 0; let currentPieces = []; let draggedPiece = null; // For drag-and-drop let selectedPiece = null; // For click-to-place const boardDiv = document.getElementById('board'); const piecesDiv = document.getElementById('pieces'); const scoreDiv = document.getElementById('score'); const highScoreDiv = document.getElementById('highScore'); const resetButton = document.getElementById('resetButton'); // Different shapes for Tetris-like pieces const pieceTemplates = [ { shape: [[1]] }, { shape: [[1, 1]] }, { shape: [[1],[1]] }, { shape: [[1,1,1]] }, { shape: [[1],[1],[1]] }, { shape: [[1,1],[1,1]] }, { shape: [[1,0],[1,0],[1,1]] }, { shape: [[1,1,1],[0,1,0]] } ]; // Colors const pieceColors = ['#e74c3c','#3498db','#f1c40f','#2ecc71','#9b59b6']; function randomChoice(arr) { return arr[Math.floor(Math.random() * arr.length)]; } // Load any saved high score function loadHighScore() { const stored = localStorage.getItem("highScore"); highScore = stored ? parseInt(stored) : 0; highScoreDiv.textContent = "High Score: " + highScore; } // Update current score, check if we beat high score function updateScore(points) { score += points; scoreDiv.textContent = "Score: " + score; if (score > highScore) { highScore = score; highScoreDiv.textContent = "High Score: " + highScore; localStorage.setItem("highScore", highScore); } } // Create an empty 8x8 board function initBoard() { board = []; for (let r = 0; r < boardSize; r++) { board[r] = []; for (let c = 0; c < boardSize; c++) { board[r][c] = null; } } renderBoard(); } // Remove preview classes from all cells function clearPreviews() { document.querySelectorAll('.cell.preview, .cell.invalid-preview, .cell.line-potential') .forEach(cell => cell.classList.remove('preview', 'invalid-preview', 'line-potential')); } // Figure out which rows/columns would be completed if we place a piece at (baseRow, baseCol) function getPotentialClearedLines(piece, baseRow, baseCol) { let completeRows = []; let completeCols = []; let rowCounts = new Array(boardSize).fill(0); let colCounts = new Array(boardSize).fill(0); // Count current blocks for (let r = 0; r < boardSize; r++) { for (let c = 0; c < boardSize; c++) { if (board[r][c]) { rowCounts[r]++; colCounts[c]++; } } } // Add blocks from the piece for (let i = 0; i < piece.shape.length; i++) { for (let j = 0; j < piece.shape[i].length; j++) { if (piece.shape[i][j] === 1) { let r = baseRow + i; let c = baseCol + j; if (r < boardSize && c < boardSize) { rowCounts[r]++; colCounts[c]++; } } } } // Identify any fully filled rows/cols for (let r = 0; r < boardSize; r++) { if (rowCounts[r] === boardSize) completeRows.push(r); } for (let c = 0; c < boardSize; c++) { if (colCounts[c] === boardSize) completeCols.push(c); } return { rows: completeRows, cols: completeCols }; } // Actually place the piece on the board function handlePiecePlacement(piece, row, col) { placePiece(piece, row, col); // Remove from current pieces currentPieces = currentPieces.filter(p => p.id !== piece.id); // Deselect if needed if (selectedPiece && selectedPiece.id === piece.id) { selectedPiece = null; } renderPieces(); renderBoard(); checkCompletedLines(); updateScore(countBlocks(piece)); // If no pieces remain, get a new set if (currentPieces.length === 0) { generatePieces(); } // Check if game over if (checkGameOver()) { setTimeout(() => alert("Game Over! Final Score: " + score), 100); } } // Draw the board as 8x8 cells function renderBoard() { boardDiv.innerHTML = ''; for (let r = 0; r < boardSize; r++) { for (let c = 0; c < boardSize; c++) { const cell = document.createElement('div'); cell.classList.add('cell'); cell.dataset.row = r; cell.dataset.col = c; cell.style.backgroundColor = board[r][c] ? board[r][c] : '#222'; // Drag events (for drag-and-drop) cell.addEventListener('dragover', e => { e.preventDefault(); clearPreviews(); if (!draggedPiece) return; const row = parseInt(cell.dataset.row); const col = parseInt(cell.dataset.col); const valid = canPlacePiece(draggedPiece, row, col); // Show previews for (let i = 0; i < draggedPiece.shape.length; i++) { for (let j = 0; j < draggedPiece.shape[i].length; j++) { if (draggedPiece.shape[i][j] === 1) { let tr = row + i; let tc = col + j; const targetCell = boardDiv.querySelector(`.cell[data-row="${tr}"][data-col="${tc}"]`); if (targetCell) { targetCell.classList.add(valid ? 'preview' : 'invalid-preview'); } } } } // Potential line clears if (valid) { const potential = getPotentialClearedLines(draggedPiece, row, col); potential.rows.forEach(rr => { for (let cc = 0; cc < boardSize; cc++) { const rowCell = boardDiv.querySelector(`.cell[data-row="${rr}"][data-col="${cc}"]`); if (rowCell) rowCell.classList.add('line-potential'); } }); potential.cols.forEach(cc => { for (let rr = 0; rr < boardSize; rr++) { const colCell = boardDiv.querySelector(`.cell[data-row="${rr}"][data-col="${cc}"]`); if (colCell) colCell.classList.add('line-potential'); } }); } }); cell.addEventListener('dragleave', clearPreviews); cell.addEventListener('drop', e => { e.preventDefault(); clearPreviews(); const row = parseInt(cell.dataset.row); const col = parseInt(cell.dataset.col); if (draggedPiece && canPlacePiece(draggedPiece, row, col)) { handlePiecePlacement(draggedPiece, row, col); } draggedPiece = null; }); // Mouseover for click-to-place preview cell.addEventListener('mouseover', () => { if (!draggedPiece && selectedPiece) { clearPreviews(); const row = parseInt(cell.dataset.row); const col = parseInt(cell.dataset.col); const valid = canPlacePiece(selectedPiece, row, col); // Show previews for (let i = 0; i < selectedPiece.shape.length; i++) { for (let j = 0; j < selectedPiece.shape[i].length; j++) { if (selectedPiece.shape[i][j] === 1) { let tr = row + i; let tc = col + j; const targetCell = boardDiv.querySelector(`.cell[data-row="${tr}"][data-col="${tc}"]`); if (targetCell) { targetCell.classList.add(valid ? 'preview' : 'invalid-preview'); } } } } // Potential line clears if (valid) { const potential = getPotentialClearedLines(selectedPiece, row, col); potential.rows.forEach(rr => { for (let cc = 0; cc < boardSize; cc++) { const rowCell = boardDiv.querySelector(`.cell[data-row="${rr}"][data-col="${cc}"]`); if (rowCell) rowCell.classList.add('line-potential'); } }); potential.cols.forEach(cc => { for (let rr = 0; rr < boardSize; rr++) { const colCell = boardDiv.querySelector(`.cell[data-row="${rr}"][data-col="${cc}"]`); if (colCell) colCell.classList.add('line-potential'); } }); } } }); cell.addEventListener('mouseleave', clearPreviews); // Click event for placing selected piece cell.addEventListener('click', () => { clearPreviews(); const row = parseInt(cell.dataset.row); const col = parseInt(cell.dataset.col); if (selectedPiece && canPlacePiece(selectedPiece, row, col)) { handlePiecePlacement(selectedPiece, row, col); } }); boardDiv.appendChild(cell); } } } // Count how many blocks in a piece function countBlocks(piece) { let count = 0; for (let row of piece.shape) { for (let val of row) { if (val === 1) count++; } } return count; } // Check if a piece can be placed at (r,c) function canPlacePiece(piece, r, c) { for (let i = 0; i < piece.shape.length; i++) { for (let j = 0; j < piece.shape[i].length; j++) { if (piece.shape[i][j] === 1) { let nr = r + i; let nc = c + j; if (nr < 0 || nr >= boardSize || nc < 0 || nc >= boardSize) return false; if (board[nr][nc]) return false; } } } return true; } // Actually fill the board array with the pieceâ€™s color function placePiece(piece, r, c) { for (let i = 0; i < piece.shape.length; i++) { for (let j = 0; j < piece.shape[i].length; j++) { if (piece.shape[i][j] === 1) { board[r + i][c + j] = piece.color; } } } } // Check for completed rows/columns, animate, then clear them function checkCompletedLines() { let rowsCleared = []; let colsCleared = []; for (let r = 0; r < boardSize; r++) { if (board[r].every(cell => cell !== null)) { rowsCleared.push(r); } } for (let c = 0; c < boardSize; c++) { let complete = true; for (let r = 0; r < boardSize; r++) { if (!board[r][c]) { complete = false; break; } } if (complete) colsCleared.push(c); } if (rowsCleared.length + colsCleared.length > 0) { // Animate fade-out rowsCleared.forEach(rr => { for (let cc = 0; cc < boardSize; cc++) { const cell = boardDiv.querySelector(`.cell[data-row="${rr}"][data-col="${cc}"]`); if (cell) cell.classList.add('line-clear'); } }); colsCleared.forEach(cc => { for (let rr = 0; rr < boardSize; rr++) { const cell = boardDiv.querySelector(`.cell[data-row="${rr}"][data-col="${cc}"]`); if (cell) cell.classList.add('line-clear'); } }); // After animation, clear them setTimeout(() => { rowsCleared.forEach(rr => { for (let cc = 0; cc < boardSize; cc++) { board[rr][cc] = null; } }); colsCleared.forEach(cc => { for (let rr = 0; rr < boardSize; rr++) { board[rr][cc] = null; } }); updateScore((rowsCleared.length + colsCleared.length) * 50); renderBoard(); }, 500); } } // Make a new set of three random pieces function generatePieces() { for (let i = 0; i < 3; i++) { const template = randomChoice(pieceTemplates); const shape = template.shape.map(row => row.slice()); currentPieces.push({ id: Date.now() + Math.random(), shape: shape, color: randomChoice(pieceColors) }); } renderPieces(); } // Show the current pieces in the #pieces area function renderPieces() { piecesDiv.innerHTML = ''; currentPieces.forEach(piece => { const pieceDiv = document.createElement('div'); pieceDiv.classList.add('piece'); if (selectedPiece && selectedPiece.id === piece.id) { pieceDiv.classList.add('selected'); } pieceDiv.draggable = true; // Start dragging pieceDiv.addEventListener('dragstart', e => { draggedPiece = piece; selectedPiece = null; renderPieces(); // Hide default drag image const crt = document.createElement('div'); e.dataTransfer.setDragImage(crt, 0, 0); }); pieceDiv.addEventListener('dragend', () => { draggedPiece = null; clearPreviews(); }); // Click to select/deselect pieceDiv.addEventListener('click', () => { selectedPiece = (selectedPiece && selectedPiece.id === piece.id) ? null : piece; renderPieces(); }); // Render the shape as a mini-grid const rows = piece.shape.length; const cols = piece.shape[0].length; pieceDiv.style.width = (cols * 22) + 'px'; pieceDiv.style.height = (rows * 22) + 'px'; pieceDiv.style.display = 'grid'; pieceDiv.style.gridTemplateColumns = `repeat(${cols}, 20px)`; pieceDiv.style.gridGap = '2px'; for (let i = 0; i < rows; i++) { for (let j = 0; j < cols; j++) { const cellDiv = document.createElement('div'); cellDiv.classList.add('piece-cell'); if (piece.shape[i][j] === 1) { cellDiv.style.backgroundColor = piece.color; } pieceDiv.appendChild(cellDiv); } } piecesDiv.appendChild(pieceDiv); }); } // Check if no piece can fit anywhere function checkGameOver() { for (let piece of currentPieces) { for (let r = 0; r < boardSize; r++) { for (let c = 0; c < boardSize; c++) { if (canPlacePiece(piece, r, c)) return false; } } } return true; } // Reset everything function resetGame() { score = 0; currentPieces = []; draggedPiece = null; selectedPiece = null; initBoard(); generatePieces(); updateScore(0); } // Hook up the Reset button resetButton.addEventListener('click', resetGame); // Start the game loadHighScore(); resetGame(); });
